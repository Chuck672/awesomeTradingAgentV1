LightweightCharts 画图工具 - 周期切换解决方案
核心问题分析
text

周期切换时的三大问题：
1. 图形消失 → 时间戳映射错误
2. 坐标不对 → 价格坐标系变化
3. 无限延伸 → 射线端点计算错误
完整解决方案

// types.ts - 核心数据结构
interface DrawingBase {
  id: string;
  type: 'horizontal_ray' | 'rectangle' | 'trend_line' | 'vertical_line';
  
  // ✅ 关键：存储【逻辑坐标】而非像素坐标
  points: LogicalPoint[];
  
  // ✅ 关键：存储【价格】而非像素Y值
  prices: number[];
  
  // 元数据
  createdAt: number;        // 创建时的Unix时间戳(ms)
  createdTimeframe: string; // 创建时的周期 '1m','5m','1h','1D'...
  
  style: DrawingStyle;
  visible: boolean;
  
  // ✅ 跨周期映射缓存（每次切换周期时更新）
  _cache?: {
    timeframe: string;
    logicals: number[];  // 当前周期的逻辑坐标
    timestamp: number;   // 缓存时间
  };
}

interface LogicalPoint {
  // 存储时间戳（毫秒），不存logical index
  // 因为logical index在不同周期完全不同
  time: number;  // Unix timestamp in milliseconds
  price: number; // 价格，跨周期不变
}

interface DrawingStyle {
  color: string;
  lineWidth: number;
  lineStyle: 'solid' | 'dashed' | 'dotted';
  fillColor?: string;
  fillOpacity?: number;
}

2. 核心坐标转换引擎
TypeScript

// coordinate-engine.ts

export class CoordinateEngine {
  private chart: IChartApi;
  private series: ISeriesApi<'Candlestick'>;
  private currentTimeframe: string;
  
  // K线数据索引：timestamp -> logical index
  private timeToLogical: Map<number, number> = new Map();
  // logical index -> timestamp
  private logicalToTime: number[] = [];
  
  constructor(
    chart: IChartApi,
    series: ISeriesApi<'Candlestick'>,
    timeframe: string
  ) {
    this.chart = chart;
    this.series = series;
    this.currentTimeframe = timeframe;
  }

  /**
   * ✅ 关键方法：更新K线数据索引
   * 每次切换周期或加载新数据后调用
   */
  updateBarIndex(bars: { time: number }[]): void {
    this.timeToLogical.clear();
    this.logicalToTime = [];
    
    bars.forEach((bar, index) => {
      // LightweightCharts的time是秒级时间戳
      const timestampMs = bar.time * 1000;
      this.timeToLogical.set(timestampMs, index);
      this.logicalToTime[index] = timestampMs;
    });
    
    console.log(`[CoordinateEngine] 更新索引: ${bars.length} 根K线, 周期: ${this.currentTimeframe}`);
  }

  /**
   * ✅ 时间戳 → 当前周期的最近logical index
   * 找不到精确匹配时，找最近的K线
   */
  timestampToLogical(timestampMs: number): number | null {
    // 1. 精确匹配
    if (this.timeToLogical.has(timestampMs)) {
      return this.timeToLogical.get(timestampMs)!;
    }
    
    // 2. 找最近的K线（二分查找）
    const times = this.logicalToTime;
    if (times.length === 0) return null;
    
    let left = 0;
    let right = times.length - 1;
    
    while (left < right) {
      const mid = Math.floor((left + right) / 2);
      if (times[mid] < timestampMs) {
        left = mid + 1;
      } else {
        right = mid;
      }
    }
    
    // 检查边界
    if (left === 0) return 0;
    if (left >= times.length) return times.length - 1;
    
    // 返回最近的
    const prevDiff = Math.abs(times[left - 1] - timestampMs);
    const currDiff = Math.abs(times[left] - timestampMs);
    
    return prevDiff <= currDiff ? left - 1 : left;
  }

  /**
   * ✅ Logical index → 时间戳
   */
  logicalToTimestamp(logical: number): number | null {
    if (logical < 0 || logical >= this.logicalToTime.length) return null;
    return this.logicalToTime[logical];
  }

  /**
   * ✅ 价格 → Y像素坐标
   */
  priceToY(price: number): number | null {
    return this.series.priceToCoordinate(price);
  }

  /**
   * ✅ Y像素坐标 → 价格
   */
  yToPrice(y: number): number | null {
    return this.series.coordinateToPrice(y);
  }

  /**
   * ✅ Logical index → X像素坐标
   */
  logicalToX(logical: number): number | null {
    const timeScale = this.chart.timeScale();
    return timeScale.logicalToCoordinate(logical);
  }

  /**
   * ✅ X像素坐标 → Logical index
   */
  xToLogical(x: number): number | null {
    const timeScale = this.chart.timeScale();
    return timeScale.coordinateToLogical(x);
  }

  /**
   * ✅ 获取当前可见区域的logical范围
   */
  getVisibleLogicalRange(): { from: number; to: number } | null {
    const timeScale = this.chart.timeScale();
    const range = timeScale.getVisibleLogicalRange();
    if (!range) return null;
    return { from: Math.floor(range.from), to: Math.ceil(range.to) };
  }

  /**
   * ✅ 获取最后一根K线的logical
   */
  getLastLogical(): number {
    return this.logicalToTime.length - 1;
  }

  /**
   * ✅ 获取第一根K线的logical
   */
  getFirstLogical(): number {
    return 0;
  }
  
  getCurrentTimeframe(): string {
    return this.currentTimeframe;
  }
  
  setCurrentTimeframe(tf: string): void {
    this.currentTimeframe = tf;
  }
}
3. 图形渲染器基类

// renderers/base-renderer.ts

export abstract class BaseRenderer {
  protected canvas: HTMLCanvasElement;
  protected ctx: CanvasRenderingContext2D;
  protected engine: CoordinateEngine;
  
  constructor(canvas: HTMLCanvasElement, engine: CoordinateEngine) {
    this.canvas = canvas;
    this.ctx = canvas.getContext('2d')!;
    this.engine = engine;
  }
  
  abstract render(drawing: DrawingBase): void;
  abstract hitTest(drawing: DrawingBase, x: number, y: number): boolean;
  
  /**
   * 通用：检查图形是否在可见区域内
   */
  protected isVisible(drawing: DrawingBase): boolean {
    const range = this.engine.getVisibleLogicalRange();
    if (!range) return false;
    
    // 获取图形的logical范围
    const logicals = drawing.points.map(p => 
      this.engine.timestampToLogical(p.time)
    ).filter(l => l !== null) as number[];
    
    if (logicals.length === 0) return false;
    
    // 对于射线，只需要起点在合理范围内
    const minLogical = Math.min(...logicals);
    const maxLogical = Math.max(...logicals);
    
    // 扩展可见范围判断（考虑射线延伸）
    return minLogical <= range.to + 100 && maxLogical >= range.from - 100;
  }
}

// renderers/horizontal-ray-renderer.ts

export class HorizontalRayRenderer extends BaseRenderer {
  
  render(drawing: DrawingBase): void {
    if (!this.isVisible(drawing)) return;
    
    const { prices, points, style } = drawing;
    
    // Horizontal Ray只需要一个价格和一个起始时间
    const price = prices[0];
    const startTime = points[0].time;
    
    // ✅ 价格转Y坐标（跨周期稳定）
    const y = this.engine.priceToY(price);
    if (y === null) return;
    
    // ✅ 起始时间转X坐标（使用最近K线匹配）
    const startLogical = this.engine.timestampToLogical(startTime);
    if (startLogical === null) return;
    
    const startX = this.engine.logicalToX(startLogical);
    if (startX === null) return;
    
    // ✅ 终点：画布右边界
    const endX = this.canvas.width;
    
    // 绘制
    this.ctx.save();
    this.ctx.strokeStyle = style.color;
    this.ctx.lineWidth = style.lineWidth;
    this.setLineStyle(style.lineStyle);
    
    this.ctx.beginPath();
    this.ctx.moveTo(startX, y);
    this.ctx.lineTo(endX, y);
    this.ctx.stroke();
    
    // 绘制价格标签
    this.renderPriceLabel(price, y, style.color);
    
    this.ctx.restore();
  }
  
  hitTest(drawing: DrawingBase, mouseX: number, mouseY: number): boolean {
    const price = drawing.prices[0];
    const y = this.engine.priceToY(price);
    if (y === null) return false;
    
    const startLogical = this.engine.timestampToLogical(drawing.points[0].time);
    if (startLogical === null) return false;
    
    const startX = this.engine.logicalToX(startLogical);
    if (startX === null) return false;
    
    // 检查X范围 & Y距离
    return mouseX >= startX && Math.abs(mouseY - y) <= 5;
  }
  
  private renderPriceLabel(price: number, y: number, color: string): void {
    const label = price.toFixed(2);
    const padding = 4;
    const textWidth = this.ctx.measureText(label).width;
    
    this.ctx.fillStyle = color;
    this.ctx.fillRect(
      this.canvas.width - textWidth - padding * 2 - 60,
      y - 10,
      textWidth + padding * 2,
      20
    );
    
    this.ctx.fillStyle = '#ffffff';
    this.ctx.font = '11px Arial';
    this.ctx.fillText(
      label,
      this.canvas.width - textWidth - padding - 60,
      y + 4
    );
  }
  
  private setLineStyle(style: string): void {
    switch (style) {
      case 'dashed':
        this.ctx.setLineDash([8, 4]);
        break;
      case 'dotted':
        this.ctx.setLineDash([2, 4]);
        break;
      default:
        this.ctx.setLineDash([]);
    }
  }
}

// renderers/rectangle-renderer.ts

export class RectangleRenderer extends BaseRenderer {
  
  render(drawing: DrawingBase): void {
    if (!this.isInViewport(drawing)) return;
    
    const { points, prices, style } = drawing;
    
    // Rectangle需要两个角点
    // points[0]: 左上角时间, prices[0]: 上边价格
    // points[1]: 右下角时间, prices[1]: 下边价格
    
    const topPrice    = Math.max(prices[0], prices[1]);
    const bottomPrice = Math.min(prices[0], prices[1]);
    const leftTime    = Math.min(points[0].time, points[1].time);
    const rightTime   = Math.max(points[0].time, points[1].time);
    
    // ✅ 价格 → Y坐标
    const topY    = this.engine.priceToY(topPrice);
    const bottomY = this.engine.priceToY(bottomPrice);
    if (topY === null || bottomY === null) return;
    
    // ✅ 时间 → X坐标（使用最近K线匹配）
    const leftLogical  = this.engine.timestampToLogical(leftTime);
    const rightLogical = this.engine.timestampToLogical(rightTime);
    
    if (leftLogical === null || rightLogical === null) return;
    
    const leftX  = this.engine.logicalToX(leftLogical);
    const rightX = this.engine.logicalToX(rightLogical);
    
    if (leftX === null || rightX === null) return;
    
    const width  = rightX - leftX;
    const height = bottomY - topY;
    
    this.ctx.save();
    
    // ✅ 填充
    if (style.fillColor) {
      this.ctx.fillStyle = style.fillColor;
      this.ctx.globalAlpha = style.fillOpacity ?? 0.1;
      this.ctx.fillRect(leftX, topY, width, height);
    }
    
    // ✅ 边框
    this.ctx.globalAlpha = 1;
    this.ctx.strokeStyle = style.color;
    this.ctx.lineWidth = style.lineWidth;
    this.ctx.strokeRect(leftX, topY, width, height);
    
    // ✅ 价格标签
    this.renderPriceLabels(topPrice, bottomPrice, topY, bottomY, rightX, style.color);
    
    this.ctx.restore();
  }
  
  hitTest(drawing: DrawingBase, mouseX: number, mouseY: number): boolean {
    const { points, prices } = drawing;
    
    const topPrice    = Math.max(prices[0], prices[1]);
    const bottomPrice = Math.min(prices[0], prices[1]);
    const leftTime    = Math.min(points[0].time, points[1].time);
    const rightTime   = Math.max(points[0].time, points[1].time);
    
    const topY    = this.engine.priceToY(topPrice);
    const bottomY = this.engine.priceToY(bottomPrice);
    
    const leftLogical  = this.engine.timestampToLogical(leftTime);
    const rightLogical = this.engine.timestampToLogical(rightTime);
    
    if (!topY || !bottomY || leftLogical === null || rightLogical === null) return false;
    
    const leftX  = this.engine.logicalToX(leftLogical);
    const rightX = this.engine.logicalToX(rightLogical);
    
    if (leftX === null || rightX === null) return false;
    
    // 检查是否在矩形边框附近（5px容差）
    const tolerance = 5;
    const inXRange = mouseX >= leftX - tolerance && mouseX <= rightX + tolerance;
    const inYRange = mouseY >= topY - tolerance && mouseY <= bottomY + tolerance;
    
    if (!inXRange || !inYRange) return false;
    
    // 边框检测
    const onLeftEdge   = Math.abs(mouseX - leftX) <= tolerance;
    const onRightEdge  = Math.abs(mouseX - rightX) <= tolerance;
    const onTopEdge    = Math.abs(mouseY - topY) <= tolerance;
    const onBottomEdge = Math.abs(mouseY - bottomY) <= tolerance;
    
    return onLeftEdge || onRightEdge || onTopEdge || onBottomEdge;
  }
  
  /**
   * ✅ 更精确的可见性检测（Rectangle专用）
   */
  private isInViewport(drawing: DrawingBase): boolean {
    const range = this.engine.getVisibleLogicalRange();
    if (!range) return false;
    
    const { points, prices } = drawing;
    
    const leftTime  = Math.min(points[0].time, points[1].time);
    const rightTime = Math.max(points[0].time, points[1].time);
    
    const leftLogical  = this.engine.timestampToLogical(leftTime);
    const rightLogical = this.engine.timestampToLogical(rightTime);
    
    if (leftLogical === null || rightLogical === null) return false;
    
    // 矩形的时间范围与可见区域有交集即显示
    return leftLogical <= range.to && rightLogical >= range.from;
  }
  
  private renderPriceLabels(
    topPrice: number,
    bottomPrice: number,
    topY: number,
    bottomY: number,
    rightX: number,
    color: string
  ): void {
    this.ctx.fillStyle = color;
    this.ctx.font = '11px Arial';
    this.ctx.fillText(topPrice.toFixed(2), rightX + 4, topY + 4);
    this.ctx.fillText(bottomPrice.toFixed(2), rightX + 4, bottomY + 4);
  }
}

// drawing-manager.ts

export class DrawingManager {
  private drawings: Map<string, DrawingBase> = new Map();
  private engine: CoordinateEngine;
  private canvas: HTMLCanvasElement;
  private ctx: CanvasRenderingContext2D;
  private currentTimeframe: string;
  
  private renderers: {
    horizontal_ray: HorizontalRayRenderer;
    rectangle: RectangleRenderer;
    // ... 其他类型
  };
  
  // 动画帧ID（避免重复渲染）
  private rafId: number | null = null;
  
  constructor(
    chart: IChartApi,
    series: ISeriesApi<'Candlestick'>,
    canvas: HTMLCanvasElement,
    timeframe: string
  ) {
    this.canvas = canvas;
    this.ctx = canvas.getContext('2d')!;
    this.currentTimeframe = timeframe;
    
    this.engine = new CoordinateEngine(chart, series, timeframe);
    
    this.renderers = {
      horizontal_ray: new HorizontalRayRenderer(canvas, this.engine),
      rectangle: new RectangleRenderer(canvas, this.engine),
    };
    
    // 监听图表滚动/缩放，触发重绘
    chart.timeScale().subscribeVisibleLogicalRangeChange(() => {
      this.scheduleRender();
    });
  }

  /**
   * ✅ 关键：切换周期时调用
   */
  async onTimeframeChange(
    newTimeframe: string,
    newBars: { time: number }[],
    newSeries: ISeriesApi<'Candlestick'>
  ): Promise<void> {
    console.log(`[DrawingManager] 切换周期: ${this.currentTimeframe} → ${newTimeframe}`);
    
    this.currentTimeframe = newTimeframe;
    
    // 1. 更新引擎配置
    this.engine.setCurrentTimeframe(newTimeframe);
    
    // 2. ✅ 更新K线索引（最关键的步骤）
    this.engine.updateBarIndex(newBars);
    
    // 3. 清除所有坐标缓存
    this.clearAllCaches();
    
    // 4. 重新渲染
    this.scheduleRender();
    
    console.log(`[DrawingManager] 周期切换完成，当前图形数量: ${this.drawings.size}`);
  }

  /**
   * 添加图形（统一入口）
   */
  addDrawing(drawing: DrawingBase): void {
    this.drawings.set(drawing.id, drawing);
    this.scheduleRender();
  }

  /**
   * 删除图形
   */
  removeDrawing(id: string): void {
    this.drawings.delete(id);
    this.scheduleRender();
  }

  /**
   * 创建 Horizontal Ray
   */
  createHorizontalRay(
    clickX: number,
    clickY: number,
    style: DrawingStyle
  ): DrawingBase | null {
    
    // ✅ 从像素坐标提取逻辑坐标
    const price = this.engine.yToPrice(clickY);
    const logical = this.engine.xToLogical(clickX);
    
    if (price === null || logical === null) return null;
    
    const timestamp = this.engine.logicalToTimestamp(logical);
    if (timestamp === null) return null;
    
    const drawing: DrawingBase = {
      id: this.generateId(),
      type: 'horizontal_ray',
      points: [{ time: timestamp, price }],  // ✅ 存时间戳
      prices: [price],                         // ✅ 存价格
      createdAt: Date.now(),
      createdTimeframe: this.currentTimeframe,
      style,
      visible: true,
    };
    
    this.addDrawing(drawing);
    return drawing;
  }

  /**
   * 创建 Rectangle
   */
  createRectangle(
    startX: number, startY: number,
    endX: number,   endY: number,
    style: DrawingStyle
  ): DrawingBase | null {
    
    const startPrice = this.engine.yToPrice(startY);
    const endPrice   = this.engine.yToPrice(endY);
    
    const startLogical = this.engine.xToLogical(startX);
    const endLogical   = this.engine.xToLogical(endX);
    
    if (
      startPrice === null || endPrice === null ||
      startLogical === null || endLogical === null
    ) return null;
    
    const startTimestamp = this.engine.logicalToTimestamp(startLogical);
    const endTimestamp   = this.engine.logicalToTimestamp(endLogical);
    
    if (startTimestamp === null || endTimestamp === null) return null;
    
    const drawing: DrawingBase = {
      id: this.generateId(),
      type: 'rectangle',
      points: [
        { time: startTimestamp, price: startPrice },  // ✅ 存时间戳+价格
        { time: endTimestamp,   price: endPrice   },
      ],
      prices: [startPrice, endPrice],
      createdAt: Date.now(),
      createdTimeframe: this.currentTimeframe,
      style,
      visible: true,
    };
    
    this.addDrawing(drawing);
    return drawing;
  }

  /**
   * 调度渲染（避免同帧多次渲染）
   */
  private scheduleRender(): void {
    if (this.rafId !== null) {
      cancelAnimationFrame(this.rafId);
    }
    this.rafId = requestAnimationFrame(() => {
      this.render();
      this.rafId = null;
    });
  }

  /**
   * 主渲染循环
   */
  private render(): void {
    // 清空画布
    this.ctx.clearRect(0, 0, this.canvas.width, this.canvas.height);
    
    // 按创建时间排序渲染
    const sortedDrawings = Array.from(this.drawings.values())
      .filter(d => d.visible)
      .sort((a, b) => a.createdAt - b.createdAt);
    
    for (const drawing of sortedDrawings) {
      try {
        this.renderDrawing(drawing);
      } catch (e) {
        console.error(`[DrawingManager] 渲染图形失败: ${drawing.id}`, e);
      }
    }
  }

  private renderDrawing(drawing: DrawingBase): void {
    switch (drawing.type) {
      case 'horizontal_ray':
        this.renderers.horizontal_ray.render(drawing);
        break;
      case 'rectangle':
        this.renderers.rectangle.render(drawing);
        break;
    }
  }

  /**
   * 清除坐标缓存
   */
  private clearAllCaches(): void {
    this.drawings.forEach(drawing => {
      delete drawing._cache;
    });
  }

  private generateId(): string {
    return `drawing_${Date.now()}_${Math.random().toString(36).substr(2, 9)}`;
  }
}
7. 数据序列化/持久化

// storage.ts

export class DrawingStorage {
  private storageKey: string;
  
  constructor(chartId: string) {
    this.storageKey = `drawings_${chartId}`;
  }
  
  save(drawings: DrawingBase[]): void {
    // ✅ 直接序列化，因为存的是时间戳和价格，跨周期安全
    const data = JSON.stringify(drawings);
    localStorage.setItem(this.storageKey, data);
  }
  
  load(): DrawingBase[] {
    const data = localStorage.getItem(this.storageKey);
    if (!data) return [];
    
    try {
      const drawings = JSON.parse(data) as DrawingBase[];
      // 清除可能存在的旧缓存
      return drawings.map(d => ({ ...d, _cache: undefined }));
    } catch (e) {
      console.error('[DrawingStorage] 加载失败', e);
      return [];
    }
  }
  
  clear(): void {
    localStorage.removeItem(this.storageKey);
  }
}

  private currentTimeframe: string = '1h';

    // 1. 创建图表
    const container = document.getElementById('chart')!;
    this.chart = createChart(container, {
      width: container.clientWidth,
      height: 600,
    });
    
    this.series = this.chart.addCandlestickSeries();
    
    // 2. 创建覆盖Canvas（与图表同尺寸）
    this.overlayCanvas = this.createOverlayCanvas(container);
    
    // 3. 加载初始数据
    const bars = await this.fetchBars(this.currentTimeframe);
    this.series.setData(bars);
    
    // 4. 初始化管理器
    this.drawingManager = new DrawingManager(
      this.chart,
      this.series,
      this.overlayCanvas,
      this.currentTimeframe
    );
    
    // ✅ 5. 初始化时必须先更新K线索引
    this.drawingManager['engine'].updateBarIndex(
      bars.map(b => ({ time: b.time * 1000 }))
    );
    
    // 6. 加载已存图形
    this.storage = new DrawingStorage('my-chart');
    const savedDrawings = this.storage.load();
    savedDrawings.forEach(d => this.drawingManager.addDrawing(d));
    
    // 7. 绑定事件
    this.bindEvents();
  }
  
  /**
   * ✅ 切换周期的完整流程
   */
  async switchTimeframe(newTimeframe: string): Promise<void> {
    if (newTimeframe === this.currentTimeframe) return;
    
    console.log(`切换周期: ${this.currentTimeframe} → ${newTimeframe}`);
    
    // 1. 获取新周期数据
    const newBars = await this.fetchBars(newTimeframe);
    
    // 2. 更新图表数据
    this.series.setData(newBars);
    
    // 3. ✅ 通知DrawingManager（必须在setData之后）
    await this.drawingManager.onTimeframeChange(
      newTimeframe,
      newBars.map(b => ({ time: b.time * 1000 })),
      this.series
    );
    
    this.currentTimeframe = newTimeframe;
    
    // 4. 保存当前图形
    this.saveDrawings();
  }
  
  private createOverlayCanvas(container: HTMLElement): HTMLCanvasElement {
    const canvas = document.createElement('canvas');
    canvas.width = container.clientWidth;
    canvas.height = container.offsetHeight;
    canvas.style.position = 'absolute';
    canvas.style.top = '0';
    canvas.style.left = '0';
    canvas.style.pointerEvents = 'none'; // 不拦截图表事件
    container.style.position = 'relative';
    container.appendChild(canvas);
    return canvas;
  }
  
  private bindEvents(): void {
    // 画布鼠标事件（用于画图）
    this.overlayCanvas.style.pointerEvents = 'auto';
    this.overlayCanvas.addEventListener('click', (e) => {
      const rect = this.overlayCanvas.getBoundingClientRect();
      const x = e.clientX - rect.left;
      const y = e.clientY - rect.top;
      
      // 根据当前工具创建图形
      this.handleCanvasClick(x, y);
    });
    
    // 周期切换按钮
    document.querySelectorAll('.timeframe-btn').forEach(btn => {
      btn.addEventListener('click', () => {
        const tf = (btn as HTMLElement).dataset.timeframe!;
        this.switchTimeframe(tf);
      });
    });
    
    // 窗口resize
    window.addEventListener('resize', () => {
      this.resizeCanvas();
    });
  }
  
  private resizeCanvas(): void {
    const container = this.overlayCanvas.parentElement!;
    this.overlayCanvas.width = container.clientWidth;
    this.overlayCanvas.height = container.offsetHeight;
    // 触发重绘
    this.drawingManager['scheduleRender']();
  }
  
  private saveDrawings(): void {
    const drawings = Array.from(this.drawingManager['drawings'].values());
    this.storage.save(drawings);
  }
  
  private async fetchBars(timeframe: string): Promise<any[]> {
    // 你的数据获取逻辑
    const response = await fetch(`/api/bars?tf=${timeframe}`);
    return response.json();
  }

    // 根据当前激活工具处理
  }
